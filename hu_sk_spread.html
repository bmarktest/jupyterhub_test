def spancreator(dataframe, colname): 
    vspan_local = [{'x0': index, 'x1': index+pd.Timedelta('1 days'), 'color':'rgba(21,14,14,0.3)','fill':True, 'opacity':.8, 'width': 0} for index, row in dataframe.iterrows() if row[colname]=='1']
    return vspan_local

Kritériumok kialakítása¶

A szlovák és magyar másnapi árak közötti spread megugrása jellemzően több hatás eredőjeként jelentkezik. Ezek vizsgálata során arra jutottunk, hogy jelentős spread-ek a következő esetekben fordulnak elő:

  1. a. a német megújulók értéke 10.000 MWh/h feletti,
  2. b. ÉS a magyar hőmérséklet 5 °C fok alatti,
  3. c. ÉS a román/szerb vízhozam 5.000 m3/s alatti (Vaskapu),
  4. d. ÉS valamilyen extrém érték teljesül a változókra: 
                                                  i.     VAGY a német megújulók értéke 20.000 MWh/h feletti is,
                                            ii.     VAGY a magyar hőmérséklet -5 °C fok alatti is,
                                           iii.     VAGY a román vízhozam 3.000 m3/s alatti,
                                           iv.     VAGY a magyar erőműi kiesés 1.000 MWh/h feletti.

df['Spread'] = df['Price_HUPX']-df['Price_ISOT']
df['German_renewables'] = df['German_solar']+df['German_wind']


criteria_a = (df['German_renewables']>10000)
criteria_b = (df['Hungary_Mean_Temp']<5)
criteria_c = (df['Bazias_Inflow']<5000)

criteria_d_i = (df['German_renewables']>20000)
criteria_d_ii = (df['Hungary_Mean_Temp']<-5)
criteria_d_iii = (df['Bazias_Inflow']<3000)

df['Big_spread'] = np.where(criteria_a & criteria_b &criteria_c & (criteria_d_i | criteria_d_ii | criteria_d_iii), '1', '0')
